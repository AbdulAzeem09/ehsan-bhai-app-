    function loadDoc() {
        var all_location_id = document.querySelectorAll('input[name="chk[]"]:checked');

        var aIds = [];
        for (var x = 0, l = all_location_id.length; x < l; x++) {
            aIds.push(all_location_id[x].value);
        }
        var str = aIds.join(', ');
        alert(str);
        var formData = new FormData();
        formData.append('visitor_id', bible_worker);
        formData.append('bible_worker_id', bible_worker);
        const xhttp = new XMLHttpRequest();
        xhttp.onload = function() {
            location.reload();
        }
        xhttp.open("POST", "<?= base_url('Assign_bibleworker/create_assign'); ?>");
        xhttp.send(formData);
    }

    function checkAll(ele) {
        var checkboxes = document.getElementsByTagName('input');
        if (ele.checked) {
            for (var i = 0; i < checkboxes.length; i++) {
                if (checkboxes[i].type == 'checkbox') {
                    checkboxes[i].checked = true;
                }
            } 
        } else {
            for (var i = 0; i < checkboxes.length; i++) {
                console.log(i)
                if (checkboxes[i].type == 'checkbox') {
                    checkboxes[i].checked = false;
                }
            }
        }
    }
